23template <typename Real>
24inline Real wigner_uppercase_d_matrix_pi_half(Real f, Real m_initial, Real m_final) {
25 static_assert(std::is_floating_point_v<Real>);
26 assert(2 * f == std::floor(2 * f) && f >= 0);
27 assert(2 * m_initial == std::floor(2 * m_initial) && m_initial >= -f && m_initial <= f);
28 assert(2 * m_final == std::floor(2 * m_final) && m_final >= -f && m_final <= f);
29
30 Real result = 0;
31 for (int k = std::max(0, static_cast<int>(m_final - m_initial));
32 k <= f + std::min(m_final, -m_initial); ++k) {
33 result += std::pow(-1, k) * maths::binomial_coefficient(f + m_final, static_cast<Real>(k)) *
34 maths::binomial_coefficient(f - m_final, k + m_initial - m_final);
35 }
36 result *= std::pow(-1, m_initial - m_final) / std::pow(2, f) *
37 std::sqrt(maths::factorial(f + m_initial) * maths::factorial(f - m_initial) /
38 (maths::factorial(f + m_final) * maths::factorial(f - m_final)));
39 return result;
40}

44template <typename Scalar>
45inline Scalar wigner_uppercase_d_matrix(typename traits::NumTraits<Scalar>::real_t f,
46 typename traits::NumTraits<Scalar>::real_t m_initial,
47 typename traits::NumTraits<Scalar>::real_t m_final,
48 typename traits::NumTraits<Scalar>::real_t alpha,
49 typename traits::NumTraits<Scalar>::real_t beta,
50 typename traits::NumTraits<Scalar>::real_t gamma) {
51 static_assert(traits::NumTraits<Scalar>::from_floating_point_v);
52 assert(2 * f == std::floor(2 * f) && f >= 0);
53 assert(2 * m_initial == std::floor(2 * m_initial) && m_initial >= -f && m_initial <= f);
54 assert(2 * m_final == std::floor(2 * m_final) && m_final >= -f && m_final <= f);
55
56 auto scale = std::max(std::abs(m_initial), std::abs(m_final)) *
57 std::max(std::abs(alpha), std::abs(gamma));
58 auto numerical_precision =
59 100 * scale * std::numeric_limits<typename traits::NumTraits<Scalar>::real_t>::epsilon();
60
61 if constexpr (traits::NumTraits<Scalar>::is_complex_v) {
62 return Scalar(std::cos(-m_initial * alpha), std::sin(-m_initial * alpha)) *
63 wigner_uppercase_d_matrix<typename traits::NumTraits<Scalar>::real_t>(
64 f, m_initial, m_final, 0, beta, 0) *
65 Scalar(std::cos(-m_final * gamma), std::sin(-m_final * gamma));
66 } else {
67 if (std::abs(
68 std::remainder(m_initial * alpha, PI<typename traits::NumTraits<Scalar>::real_t>)) >
69 numerical_precision) {
70 throw std::invalid_argument(
71 "The scalar type must be complex if m_initial*alpha is not a multiple of pi");
72 }
73 if (std::abs(
74 std::remainder(m_final * gamma, PI<typename traits::NumTraits<Scalar>::real_t>)) >
75 numerical_precision) {
76 throw std::invalid_argument(
77 "The scalar type must be complex if m_final*gamma is not a multiple of pi");
78 }
79 std::complex<Scalar> result = 0;
80 for (typename traits::NumTraits<Scalar>::real_t m = -f;
81 m <= f; // NOSONAR m is precisely representable
82 ++m) {
83 result += wigner_uppercase_d_matrix_pi_half(f, m_initial, m) *
84 std::complex<typename traits::NumTraits<Scalar>::real_t>(std::cos(-m * beta),
85 std::sin(-m * beta)) *
86 wigner_uppercase_d_matrix_pi_half(f, m, -m_final);
87 }
88 result *= std::pow(std::complex<typename traits::NumTraits<Scalar>::real_t>(0, 1),
89 2 * f - m_initial - m_final) *
90 static_cast<typename traits::NumTraits<Scalar>::real_t>(std::pow(-1, 2 * m_initial));
91 return result.real();
92 }
